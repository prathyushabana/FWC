\section*{11$^{th}$ Maths - EXERCISE-10.4}

Find the  equations of the lines, which cutoff intercepts on the axes  whose sum and product are 1 and -6 respectively.

\solution
Let the intercepts of $x$ and $y$ are $a$ and $b$\\

\begin{table}[!h]
\centering
\input{./figs/table.tex}
\caption{}
\label{Inputs}
\end{table}
Given
\begin{align}
a+b&=1
\label{eq1}\\
ab&=-6
\label{eq2}
\end{align} 
on solving \eqref{eq1} and \eqref{eq2} we get\\

\begin{align}
\implies a=3,b=-2 \text{ or }   a=-2,b=3
\end{align}
Thus,the possible intercepts are\\
\begin{align}
\myvec{3\\ 0 \\} ,\myvec{0\\ -2 \\} ,\myvec{-2\\ 0 \\}, \myvec{0\\ 3 \\}
\end{align}
\begin{align}
\vec{m}&=\vec{a}-\vec{b}\\
&=\myvec{3\\0}-\myvec{0\\-2}\\
&=\myvec{3\\2}\\
&=\myvec{3\\2} \text{ or } \myvec{-2\\-3}\\
\end{align}
\begin{enumerate}
\item For
\begin{align}
\vec{n} = \myvec{ 2\\-3}\\
\end{align}
the equation of the line will be\\
\begin{align}
\vec{n}^{\top}\brak{\vec{x}-\vec{A}}=0\\
\myvec{2&-3}\vec{x}=-6
\end{align}
\item For
\begin{align}
\vec{n}=\myvec{-3\\2}\\
\end{align}
the equation of the line will be\\
\begin{align}
\vec{n}^{\top}\brak{\vec{x}-\vec{B}}=0\\
\myvec{-3&2}\vec{x}=-6
\end{align}
\end{enumerate}
